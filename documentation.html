  constructor(parent) {
    super(parent)
    this.people = new ChainedSet(this)
  }

    const person = new FactoryChain(this)
    person
      .props(['name', 'age', 'email'])
      .onChainUpDown(this.person)
      .chainUpDowns(['person'])

      })

  }
}

  .person()
